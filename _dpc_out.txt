    def _deny_permit(self, plan_id: str, symbol: str, reason: str, context: Dict) -> Dict:
        """Deny P3.3 execution permit"""
        permit_key = f"quantum:permit:p33:{plan_id}"
        
        permit_data = {
            'allow': False,
            'symbol': symbol,
            'reason': reason,
